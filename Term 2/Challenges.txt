#3: Median of sorted array

  BEGIN
    arrNumbers <- sortedArray
    mid <- length(arrNumbers)/2 
    IF mid MOD 2 != 0 THEN
    //odd
      median <- arrNumbers[mid-1]
    ELSE 
    //even

      high <- arrNumbers[length(arrNumbers)/2]
      low <- arrNumbers[length(arrNumbers)/2 -1]

      median <- (high + low)/2  
    END IF

    OUTPUT median

  END
